# すべてのクローラーに対する基本ルール
# 特定のクローラーを除き、サイト全体へのアクセスを拒否します。
User-agent: *
Disallow: /

# Googlebotに対するルール
# Googleのクローラーにはサイト全体へのアクセスを許可します。
User-agent: Googlebot*
Allow: /

# bingbotに対するルール
# Bingのクローラーにもサイト全体へのアクセスを許可します。
User-agent: bingbot
Allow: /

# DuckDuckGo (DuckDuckBot)
User-agent: DuckDuckBot
Allow: /

# Yahoo! JAPAN（検索用）
User-agent: Yahoo! Slurp
Allow: /

# Internet Archive
User-agent: ia_archiver
Allow: /


# Common Crawlのクローラーに対するルール
# Common Crawlのクローラーにはサイト全体へのアクセスを許可しますが、
# /blog/以下のパスは拒否します。
User-agent: CCBot
Allow: /
Disallow: /blog/
